public class Solution {
    public void rotate(int[] nums, int k) {
        k=k%nums.length;
        int[] tmp=new int[k];
        int j=0;
        int index=0;
        for(int i=nums.length-1;i>=nums.length-k;i--){
            tmp[j++]=nums[i];
            nums[i]=i<k?tmp[index++]:nums[i-k];
        }
        for(int i=nums.length-k-1;i>=0;i--){
            nums[i]=i<k?tmp[index++]:nums[i-k];
        }
    }
}
